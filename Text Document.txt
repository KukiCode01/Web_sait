    // JavaScript для открытия и закрытия модального окна
    function openModal() {
        var modal = document.getElementById('myModal');
        if (modal) {
            modal.style.display = 'flex';
        }
    }

    function closeModal() {
        var modal = document.getElementById('myModal');
        if (modal) {
            modal.style.display = 'none';
        }
    }
